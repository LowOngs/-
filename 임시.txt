  }
  #opsBtn svg{width:18px;height:18px;fill:#fff;opacity:.9}
  .show2s{display:block;opacity:0;animation:fade .15s forwards}
  @keyframes fade{to{opacity:1}}
</style>

<div id="opsWrap" aria-hidden="true">
  <a id="opsBtn" href="#" rel="nofollow noopener" aria-label="스크립트 관리" title="스크립트 관리">
    <svg viewbox="0 0 24 24" width="18" height="18" aria-hidden="true">

      <g stroke="#fff" stroke-width="2" fill="none" stroke-linecap="round">

<script>
(function(){
  const wrap = document.getElementById('opsWrap');
  const btn  = document.getElementById('opsBtn');

  // 네 실제 배포 URL
  const ADMIN_URL_BASE = 'https://script.google.com/macros/s/AKfycbycWpb0FzgIcbs-WuIjcSGEVk8ITvNX1P99INi1US4Q2TLXrexBzPOMNPnClRosPUsl-w/exec
';
  const ADMIN_URL = ADMIN_URL_BASE + (ADMIN_URL_BASE.includes('?') ? '&' : '?') + 'cb=' + Date.now();

  let openUntil = 0;
  const DURATION = 2000; // 2초

  // Shift + A 누르면 2초간 표시/허용
  document.addEventListener('keydown', function(e){
    if(e.shiftKey && (e.key==='a' || e.key==='A')){
      wrap.classList.add('show2s');
      wrap.style.display='block';
      openUntil = Date.now() + DURATION;
      setTimeout(()=>{ wrap.style.display='none'; }, DURATION);
    }
  });

  // 버튼 클릭: 2초 내에만 동작
  btn.addEventListener('click', function(e){
    e.preventDefault();
    if (Date.now() <= openUntil) {
      window.open(ADMIN_URL, '_blank', 'noopener');
    }
  });
})();
</script>
