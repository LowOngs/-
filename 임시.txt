<script>
(function(){
  /* ⬇ 쿼리 없는 exec 주소 */
  const EXEC_BASE = 'https://script.google.com/macros/s/AKfycbxVHS_QoctY0cxMq9eHZGxlwT7obwzajAnvV22WDHJ-F6kgJDazRK6crq6mgLnjGeBuIw/exec';

  const withQS = (base, params) => {
    const u = new URL(base);
    Object.entries(params).forEach(([k,v]) => u.searchParams.set(k, v));
    return u.toString();
  };

  window.addEventListener('load', init, {once:true});

  function forceGray(el){
    el.style.filter = 'grayscale(1) brightness(.85) opacity(.55)';
    const img = el.querySelector('img'); if (img) img.style.filter = 'grayscale(1) brightness(.9) opacity(.75)';
    el.style.pointerEvents = 'none';
    el.removeAttribute('href'); el.removeAttribute('target'); el.removeAttribute('rel');
    el.classList.remove('is-active');
  }
  function forceActive(el){
    el.style.filter = ''; const img = el.querySelector('img'); if (img) img.style.filter = '';
    el.style.pointerEvents = '';
    el.classList.add('is-active');
  }

  function init(){
    const els = document.querySelectorAll('.partner-btn[data-pid]');
    if (!els.length) return;

    // 기본: 모두 회색·비클릭(인라인 강제)
    els.forEach(forceGray);

    // (보호) 비활성 클릭 차단
    document.addEventListener('click', function(e){
      const a = e.target.closest('.partner-btn');
      if (!a) return;
      if (!a.classList.contains('is-active')) { e.preventDefault(); e.stopPropagation(); }
    }, true);

    // 활성맵 로드
    fetch(withQS(EXEC_BASE, {cmd:'map'}), {cache:'no-store'})
      .then(r => { if(!r.ok) throw new Error('map http '+r.status); return r.json(); })
      .then(map => {
        els.forEach(el=>{
          const pid = el.getAttribute('data-pid');
          const active = !!(map && map[pid]);
          if (active){
            forceActive(el);
            el.href   = withQS(EXEC_BASE, {cmd:'go', pid, src:'footer'});
            el.target = '_blank';
            el.rel    = 'nofollow noopener';
          } else {
            forceGray(el);
          }
        });
      })
      .catch(err=>{
        console.warn('[partners] map fetch failed:', err);
        // 실패 시 모두 회색 유지
      });
  }
})();
</script>
