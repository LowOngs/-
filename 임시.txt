<script>
(function(){
  const wrap = document.getElementById('opsWrap');
  const btn  = document.getElementById('opsBtn');

  /* ✅ 웹앱 주소 (옹스님 실제 배포 URL로 교체) */
  const BASE_URL = 'https://script.google.com/macros/s/AKfycbzdngAqXOjZjd2_50snpGx7TfyZ_iFYABWKrBYRStDDdUjmFxOOmxnUO5xWiUr8YWDrKw/exec';

  /* ✅ 캐시버스터 쿼리 생성 */
  const ADMIN_URL = BASE_URL + '?v=' + Date.now();

  let openUntil = 0;
  // Shift + A 누르면 1초간 표시
  document.addEventListener('keydown', function(e){
    if(e.shiftKey && (e.key==='a' || e.key==='A')){
      wrap.classList.add('show1s');
      wrap.style.display='block';
      openUntil = Date.now() + 1000;
      setTimeout(()=>{ wrap.style.display='none'; }, 1000);
    }
  });

  // 버튼 클릭: 1초 내에만 동작
  btn.addEventListener('click', function(e){
    e.preventDefault();
    if (Date.now() <= openUntil) {
      window.open(ADMIN_URL, '_blank', 'noopener');
    }
  });
})();
</script>
