<!-- 웹앱 데이터 호출 -->
<script>
(function(){
  // ✅ 이 부분을 옹스님이 만든 Web App URL로 교체
  const API_URL = 'REPLACE_WITH_WEB_APP_URL';

  function formatNumber(num){
    try { return new Intl.NumberFormat('en-US').format(num); }
    catch(e){ return num.toString(); }
  }

  fetch(API_URL, { cache: 'no-store' })
    .then(res => res.json())
    .then(data => {
      const total = data.yearly || data.total || data.all || 0;
      const el = document.getElementById('custom_total_counter');
      if(el) el.textContent = formatNumber(total);
    })
    .catch(err => console.error('Stats load error:', err));
})();
</script>
